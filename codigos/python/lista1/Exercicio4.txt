n=1
negativo = 0
positivo = 0
while n==1:
    valor = int(input("Digite um valor:"))
    if valor<0:
        negativo = negativo + valor
    else:
        if valor>0:
            positivo = positivo + valor
        else:
            n=2
print("A soma dos valores positivos é:",positivo)
print("A soma dos valores negativos é:",negativo)
