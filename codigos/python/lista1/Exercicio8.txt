inteiros = int(input("Digite um valor inteiro:"))
n=0
while n<inteiros:
    reais = int(input("Digite um valor real:"))
    if n==0:
        maior = reais
        menor = reais
    elif reais > maior:
        maior = reais
    else:
        if reais < menor:
            menor = reais
    n=n+1
print("O menor é:",menor)
print("O maior é:",maior)
print("Fim do programa")
