n=1
while n==1:
    peso = float(input("Informe o peso do lutador:"))
    if peso <= 0:
        n = 0
    elif peso < 65:
        print("Pena")
    elif peso < 72:
        print("Leve")
    elif peso < 79:
        print("Ligeiro")
    elif peso < 86:
        print("Meio médio")
    elif peso < 93:
        print("Médio")
    elif peso < 100:
        print("Meio pesado")
    else:
        print("Pesado")
print("Fim do programa")
