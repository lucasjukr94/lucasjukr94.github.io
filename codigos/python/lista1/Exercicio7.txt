min = int(input("Digite um valor mínimo do intervalo:"))
max = int(input("Digite um valor máximo do intervalo:"))
if max < min:
    print("O valor mínimo é maior que o máximo")
n=0
while n<=max-min:
    valor = min + n
    if valor%5 == 0:
        print(valor)
    n = n+1
print("Fim do programa")
